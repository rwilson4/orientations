fn angle_axis(&self) -> (f64, Vector3d)
[src]

Get the angle and axis associated with a rotation. If the rotation is the identity, the z-axis will be returned.

Examples

use orientations::*;
let q = Quaternion::identity();
let (angle, axis) = q.angle_axis();
assert_eq!(angle, 0.0);
assert_eq!(axis, Vector3d::z());
